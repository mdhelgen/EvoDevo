00006 // The Mersenne Twister is an algorithm for generating random numbers.  It
00007 // was designed with consideration of the flaws in various other generators.
00008 // The period, 2^19937-1, and the order of equidistribution, 623 dimensions,
00009 // are far greater.  The generator is also fast; it avoids multiplication and
00010 // division, and it benefits from caches and pipelines.  For more information
00011 // see the inventors' web page at
00012 // http://www.math.sci.hiroshima-u.ac.jp/~m-mat/MT/emt.html

00142 inline MTRand::uint32 MTRand::hash( time_t t, clock_t c )
00143 {
00144         // Get a uint32 from t and c
00145         // Better than uint32(x) in case x is floating point in [0,1]
00146         // Based on code by Lawrence Kirby (fred@genesis.demon.co.uk)
00147         
00148         static uint32 differ = 0;  // guarantee time-based seeds will change
00149         
00150         uint32 h1 = 0;
00151         unsigned char *p = (unsigned char *) &t;
00152         for( size_t i = 0; i < sizeof(t); ++i )
00153         {
00154                 h1 *= UCHAR_MAX + 2U;
00155                 h1 += p[i];
00156         }
00157         uint32 h2 = 0;
00158         p = (unsigned char *) &c;
00159         for( size_t j = 0; j < sizeof(c); ++j )
00160         {
00161                 h2 *= UCHAR_MAX + 2U;
00162                 h2 += p[j];
00163         }
00164         return ( h1 + differ++ ) ^ h2;
00165 }

00167 inline void MTRand::initialize( const uint32 seed )
00168 {
00169         // Initialize generator state with seed
00170         // See Knuth TAOCP Vol 2, 3rd Ed, p.106 for multiplier.
00171         // In previous versions, most significant bits (MSBs) of the seed affect
00172         // only MSBs of the state array.  Modified 9 Jan 2002 by Makoto Matsumoto.
00173         register uint32 *s = state;
00174         register uint32 *r = state;
00175         register int i = 1;
00176         *s++ = seed & 0xffffffffUL;
00177         for( ; i < N; ++i )
00178         {
00179                 *s++ = ( 1812433253UL * ( *r ^ (*r >> 30) ) + i ) & 0xffffffffUL;
00180                 r++;
00181         }
00182 }

00184 inline void MTRand::reload()
00185 {
00186         // Generate N new values in state
00187         // Made clearer and faster by Matthew Bellew (matthew.bellew@home.com)
00188         static const int MmN = int(M) - int(N);  // in case enums are unsigned
00189         register uint32 *p = state;
00190         register int i;
00191         for( i = N - M; i--; ++p )
00192                 *p = twist( p[M], p[0], p[1] );
00193         for( i = M; --i; ++p )
00194                 *p = twist( p[MmN], p[0], p[1] );
00195         *p = twist( p[MmN], p[0], state[0] );
00196         
00197         left = N, pNext = state;
00198 }

00207 inline void MTRand::seed( uint32 *const bigSeed, const uint32 seedLength )
00208 {
00209         // Seed the generator with an array of uint32's
00210         // There are 2^19937-1 possible initial states.  This function allows
00211         // all of those to be accessed by providing at least 19937 bits (with a
00212         // default seed length of N = 624 uint32's).  Any bits above the lower 32
00213         // in each element are discarded.
00214         // Just call seed() if you want to get array from /dev/urandom
00215         initialize(19650218UL);
00216         register int i = 1;
00217         register uint32 j = 0;
00218         register int k = ( N > seedLength ? N : seedLength );
00219         for( ; k; --k )
00220         {
00221                 state[i] =
00222                 state[i] ^ ( (state[i-1] ^ (state[i-1] >> 30)) * 1664525UL );
00223                 state[i] += ( bigSeed[j] & 0xffffffffUL ) + j;
00224                 state[i] &= 0xffffffffUL;
00225                 ++i;  ++j;
00226                 if( i >= N ) { state[0] = state[N-1];  i = 1; }
00227                 if( j >= seedLength ) j = 0;
00228         }
00229         for( k = N - 1; k; --k )
00230         {
00231                 state[i] =
00232                 state[i] ^ ( (state[i-1] ^ (state[i-1] >> 30)) * 1566083941UL );
00233                 state[i] -= i;
00234                 state[i] &= 0xffffffffUL;
00235                 ++i;
00236                 if( i >= N ) { state[0] = state[N-1];  i = 1; }
00237         }
00238         state[0] = 0x80000000UL;  // MSB is 1, assuring non-zero initial array
00239         reload();
00240 }

00242 inline void MTRand::seed()
00243 {
00244         // Seed the generator with an array from /dev/urandom if available
00245         // Otherwise use a hash of time() and clock() values
00246         
00247         // First try getting an array from /dev/urandom
00248         FILE* urandom = fopen( "/dev/urandom", "rb" );
00249         if( urandom )
00250         {
00251                 uint32 bigSeed[N];
00252                 register uint32 *s = bigSeed;
00253                 register int i = N;
00254                 register bool success = true;
00255                 while( success && i-- )
00256                         success = fread( s++, sizeof(uint32), 1, urandom );
00257                 fclose(urandom);
00258                 if( success ) { seed( bigSeed, N );  return; }
00259         }
00260         
00261         // Was not successful, so use time() and clock() instead
00262         seed( hash( time(NULL), clock() ) );
00263 }

00284 inline MTRand::uint32 MTRand::randInt()
00285 {
00286         // Pull a 32-bit integer from the generator state
00287         // Every other access function simply transforms the numbers extracted here
00288         
00289         if( left == 0 ) reload();
00290         --left;
00291         
00292         register uint32 s1;
00293         s1 = *pNext++;
00294         s1 ^= (s1 >> 11);
00295         s1 ^= (s1 <<  7) & 0x9d2c5680UL;
00296         s1 ^= (s1 << 15) & 0xefc60000UL;
00297         return ( s1 ^ (s1 >> 18) );
00298 }

00300 inline MTRand::uint32 MTRand::randInt( const uint32 n )
00301 {
00302         // Find which bits are used in n
00303         // Optimized by Magnus Jonsson (magnus@smartelectronix.com)
00304         uint32 used = n;
00305         used |= used >> 1;
00306         used |= used >> 2;
00307         used |= used >> 4;
00308         used |= used >> 8;
00309         used |= used >> 16;
00310         
00311         // Draw numbers until one is found in [0,n]
00312         uint32 i;
00313         do
00314                 i = randInt() & used;  // toss unused bits to shorten search
00315         while( i > n );
00316         return i;
00317 }

00343 inline double MTRand::randNorm( const double mean, const double stddev )
00344 {
00345         // Return a real number from a normal (Gaussian) distribution with given
00346         // mean and standard deviation by polar form of Box-Muller transformation
00347         double x, y, r;
00348         do
00349         {
00350                 x = 2.0 * rand() - 1.0;
00351                 y = 2.0 * rand() - 1.0;
00352                 r = x * x + y * y;
00353         }
00354         while ( r >= 1.0 || r == 0.0 );
00355         double s = sqrt( -2.0 * log(r) / r );
00356         return mean + x * s * stddev;
00357 }
